Notification System - ES Module

Ejemplo de uso con ES6 Modules (import/export) 

<script type="module">
  // Importar desde jsDelivr CDN
  import NotificationSystem from 'https://cdn.jsdelivr.net/gh/Fernandocabal/fernotify@latest/dist/notification-system.esm.js';
  
  // Crear instancia global
  window.notify = new NotificationSystem();
  
  // Usar
  notify.success('¡Funcionando con ES Modules!');
</script>
